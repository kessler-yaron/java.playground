
---[ Java Arithmetic - Assignment Operators - Short hand ]------------

 (number == 10) ==> (number = number + 2) ==>   (number += 2) = 12
 (number == 12) ==> (number = number - 3) ==>   (number -= 3) = 9
 (number == 9) ==> (number = number * 4) ==>   (number *= 4) = 36
 (number == 36) ==> (number = number / 3) ==>   (number /= 3) = 12
 (number == 12) ==> (number = number % 5) ==>   (number %= 4) = 2
